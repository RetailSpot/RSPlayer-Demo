Format is show at the bottom right of the page.
It can be collapsed by clicking on the minus sign (-).

Add the monna slider in your page using this code in a <script> tag.
You can find the full library documentation for outstream formats Here 
				
var s = document.createElement("script");
s.src = "https://static.retailspotads.com/rsplayer.js";
s.onload = function () {
	const RS = window.RetailSpot;

	const RetailSpotConfig = {
		placement: 'eq-680731-2028801-135282',
		partnerId: '644250',
		width: 200,
		height: 340,
		format: 'slider',
		hideButtons: 3000,
		autoroll: 6000,
		collapseButton: true,
		widget: "monna",
		mobile: {
			width: 360,
			height: 290,
		}
	}

	RS.init(RetailSpotConfig, (ad) => {
		if (ad.subscribe){
			ad.subscribe(() => RS.tools.track("%%CLICK_URL_UNESC%%"), RS.VPAID_EVENTS.AdClickThru);
		}
	});
}
document.body.appendChild(s);
