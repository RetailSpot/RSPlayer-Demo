Oustream video/carousel in Monna widget

Add the RetailSpot player in your page using this code.
You can find the full library documentation for outstream formats Here 
        
<div id="target"></div>
<script type="text/javascript">
 (function () {
  const targetElement = document.currentScript.parentNode;
  const s = document.createElement("script");
  s.src = "https://static.retailspotads.com/rsplayer.js";
  s.onload = function () {
   const RS = window.RetailSpot;
   
   const RetailSpotConfig = {
      placement: "68c04b493a9a482fc8053470",
      positionSelector: "#target",
      itemsMinSize: 200,
      autoroll: 4000,
      arrowAbove: true,
      width: "100%",
      height: 300,
      videoLoop: true,
      hideButtons: 3000,
      closeButton: false,
      secondaryColor: "#f9c863",
      widget: "monna",
    };
  
   RS.init(RetailSpotConfig, (ad) => {
    if (ad && ad.subscribe){
      ad.subscribe(() => {
        console.log("AdError event received");
      }, RS.VPAID_EVENTS.AdError); 
    }
   });
  };
  document.body.appendChild(s);
 })();
</script>
