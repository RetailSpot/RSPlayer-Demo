Add the RetailSpot player in your page using this code.
You can find the full library documentation for outstream formats Here 
        
<div id="target"></div>
<script type="text/javascript">
 (function () {
  const targetElement = document.currentScript.parentNode;
  const s = document.createElement("script");
  s.src = "https://static.retailspotads.com/rsplayer.js";
  s.onload = function () {
   const RS = window.RetailSpot;
   const template = `<div style="border: solid #004196;border-width: 2px 1px;width:100%;height:100%;background:#fff;overflow:hidden;display:flex;flex-direction:column;align-items:flex-start;padding:10px;box-sizing:border-box;font-family:sans-serif;">
    <img class="rs-el-img" style="width:200px;height:160px;object-fit:contain;margin-bottom:10px;">
    <div class="rs-el-title" style="max-height: 50px;font-size:14px;font-weight:bold;color:#004196;text-align:left;margin: 0 0 5px 20px;"></div>
    <div class="rs-el-description" style="font-size:14px;text-align:left;margin: 0 20px 10px 20px;min-height: 32px;"></div>
    <div class="rs-el-price" style="color:#004196;font-size:23px;font-weight:600;text-align:left;margin: 0 20px 10px 20px;flex-grow:1;"></div>
</div>
`;
   const RetailSpotConfig = {
      placement: "68c04b493a9a482fc8053470",
      positionSelector: "#target",
      itemsMinSize: 200,
      autoroll: 4000,
      arrowAbove: true,
      width: "100%",
      height: 300,
      videoLoop: true,
      hideButtons: 3000,
      closeButton: false,
      secondaryColor: "#f9c863",
      widget: "side-by-side",
      template: template
    };
  
   RS.init(RetailSpotConfig, (ad) => {
    if (ad && ad.subscribe){
      ad.subscribe(() => {
        console.log("AdError event received");
      }, RS.VPAID_EVENTS.AdError); 
    }
   });
  };
  document.body.appendChild(s);
 })();
</script>
